\begin{frame}{Condorcet: Exemplo Numérico ($n=5$, $p=0{,}6$)}
    \begin{block}{Cálculo}
        \[
        P_5 = \sum_{k=3}^{5} \binom{5}{k} \cdot 0{,}6^k \cdot 0{,}4^{5-k}
        \]
    \end{block}

    \vspace{0.15cm}

    \begin{itemize}
        \item $k=3$: $\binom{5}{3} \cdot 0{,}6^3 \cdot 0{,}4^2 = 10 \times 0{,}216 \times 0{,}16 = 0{,}3456$
        \item $k=4$: $\binom{5}{4} \cdot 0{,}6^4 \cdot 0{,}4^1 = 5 \times 0{,}1296 \times 0{,}4 = 0{,}2592$
        \item $k=5$: $\binom{5}{5} \cdot 0{,}6^5 \cdot 0{,}4^0 = 1 \times 0{,}07776 \times 1 = 0{,}07776$
    \end{itemize}

    \vspace{0.15cm}

    \begin{exampleblock}{Resultado}
        \[
        P_5 = 0{,}3456 + 0{,}2592 + 0{,}07776 = 0{,}68256 \approx 68\%
        \]
    \end{exampleblock}

    \simplebox{Com apenas 5 votantes e $p = 0{,}6$, a maioria acerta \textbf{68\%} das vezes --- superior aos 60\% individuais.}
\end{frame}

\begin{frame}{Centralização: Escolha Política}
    \begin{alertblock}{Fato Histórico}
        A Bolsa de Amsterdã funcionou por \textbf{mais de um século} antes de regulamentações significativas serem impostas \cite{dejong2005}.
    \end{alertblock}

    \vspace{0.3cm}

    \begin{itemize}
        \item A centralização posterior foi \textbf{escolha política}, não necessidade técnica
        \item Regulamentações frequentemente atendiam interesses de grupos específicos
        \item Os mercados de capitais \textbf{nasceram descentralizados}
    \end{itemize}

    \vspace{0.3cm}

    \successbox{A blockchain não propõe algo novo --- oferece ferramentas para \textbf{retornar aos princípios fundacionais} do mercado de capitais.}
\end{frame}

\subsection{Tecnologia Blockchain}
\begin{frame}{Fundamentos de Blockchain}
    \begin{block}{Características Principais}
        \begin{itemize}
            \item \textbf{Descentralização}: Dados replicados em múltiplos nós
            \item \textbf{Imutabilidade}: Transações não podem ser alteradas
            \item \textbf{Transparência}: Regras públicas e verificáveis
            \item \textbf{Segurança criptográfica}: Proteção por algoritmos robustos
        \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Inovação Central}
        Resolve o \textbf{Problema dos Generais Bizantinos}: múltiplos participantes alcançam consenso sem autoridade central.
    \end{alertblock}
\end{frame}

%% ---------------------------------------------------------------------------
\begin{frame}{O Problema dos Generais Bizantinos}
    \begin{block}{Formulação \cite{lamport1982}}
        Múltiplos generais cercando uma cidade devem coordenar um ataque simultâneo. Alguns podem ser \textbf{traidores} enviando mensagens contraditórias.
    \end{block}

    \vspace{0.2cm}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{O Desafio}
                \begin{itemize}
                    \item Como alcançar \textbf{consenso} com traidores na rede?
                    \item Considerado \textbf{impossível} em redes abertas por décadas
                \end{itemize}
            \end{alertblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Solução: Bitcoin (2008)}
                \begin{itemize}
                    \item \textbf{Prova de Trabalho} (Hashcash \cite{back2002})
                    \item Cadeia de blocos com hash do anterior
                    \item Regra: ``maior trabalho acumulado''
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \vspace{0.2cm}

    \simplebox{O Bitcoin resolveu na prática: consenso \textbf{sem autoridade central}, usando funções hash e prova de trabalho.}
\end{frame}

\subsection{Criptografia}
\begin{frame}{Funções Hash Criptográficas}
    \begin{block}{Definição}
        Função unidirecional $H(m) = x$ que mapeia entrada de tamanho arbitrário para saída de tamanho fixo \cite{tomaz2014}.
    \end{block}

    \vspace{0.2cm}

    \begin{table}
        \centering
        \small
        \begin{tabular}{p{4cm}p{5.5cm}}
            \toprule
            \textbf{Propriedade} & \textbf{Significado} \\
            \midrule
            Resistência à pré-imagem & Dado $x$, inviável encontrar $m$ \\
            Res.\ à 2ª pré-imagem & Dado $m$, inviável achar $m' \neq m$ com $H(m')=H(m)$ \\
            Resistência à colisão & Inviável achar \textit{qualquer} par $(m, m')$ colidindo \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.2cm}

    \begin{exampleblock}{Paradoxo do Aniversário}
        $\bar{p}(n) = \dfrac{365!}{365^n \cdot (365-n)!}$ \quad --- \quad Com 23 pessoas, $P(\text{colisão}) > 50\%$.
        Para SHA-256: necessárias $2^{128}$ tentativas (ataque do aniversário).
    \end{exampleblock}
\end{frame}

                    Pré-imagem & $2^n$ \\
                    2ª pré-imagem & $2^n$ \\
                    Colisão & $2^{n/2}$ \\
                    \bottomrule
                \end{tabular}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}

    \begin{alertblock}{Algoritmo de Yuval (1979)}
        Gera $2^{n/2}$ variações de $m_1$ e $m_2$, busca $H(m'_1) = H(m'_2)$. Para SHA-256: $2^{128}$ tentativas --- \textbf{computacionalmente inviável} \cite{yuval1979}.
    \end{alertblock}
\end{frame}

\begin{frame}{Criptossistema RSA}
    \begin{block}{Geração de Chaves (5 passos) \cite{rivest1978, tomaz2014}}
        \begin{enumerate}
            \item Escolher primos grandes $p$ e $q$
            \item Calcular módulo $n = p \times q$
            \item Calcular totiente $\phi(n) = (p-1)(q-1)$
            \item Escolher $e$ tal que $\text{mdc}(e, \phi(n)) = 1$
            \item Calcular $d$: \; $e \cdot d \equiv 1 \pmod{\phi(n)}$ \; (alg.\ euclidiano estendido)
        \end{enumerate}
    \end{block}

    \vspace{0.3cm}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Cifragem / Decifragem}
                $C = M^e \bmod n$ \\[4pt]
                $M = C^d \bmod n$
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Assinatura Digital}
                \text{Assinar}: $S = H(M)^d \bmod n$ \\[4pt]
                \text{Verificar}: $H(M) = S^e \bmod n$ \\[4pt]
                {\small Garante: integridade, autenticidade, não repúdio}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{RingCT: Ocultando os Valores}
    \begin{block}{Pedersen Commitment}
        \[
        C = vG + rH
        \]
        onde $G$, $H$ são pontos geradores (públicos), $r$ é o \textit{blinding factor} (secreto).
    \end{block}

    \vspace{0.15cm}

    \begin{block}{Propriedade Homomórfica Aditiva}
        \[
        \sum C_{in} - \sum C_{out} = 0
        \]
        Verificação sem conhecer valores individuais --- dinheiro não é criado do nada.
    \end{block}

    \vspace{0.15cm}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{\footnotesize Visão do Remetente}
                {\small Entradas: 7 + 5 = 12 XMR \\
                Saídas: 9 + 3 = 12 XMR \\
                \faCheck\ $12 = 12$}
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{\footnotesize Visão da Blockchain}
                {\small $C_{in_1}, C_{in_2} \to C_{out_1}, C_{out_2}$ \\
                Valores \textbf{ocultos} \\
                \faEyeSlash\ Impossível descobrir}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}
