\section{Introdução}

%% --- Slide 1: Contextualização ---
\begin{frame}{Contextualização}
    \begin{itemize}
        \item O Brasil enfrenta \textbf{desafios econômicos estruturais} que limitam seu desenvolvimento há décadas \cite{giambiagi2011}
        \item O foco excessivo em corrupção e polarização desvia a atenção dos problemas estruturantes:
        \begin{itemize}
            \item Infraestrutura deficiente
            \item Baixa taxa de poupança nacional
            \item Crédito historicamente caro
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Problema Central}
        Os gastos com previdência e assistência social representam aproximadamente \textbf{54\% das despesas primárias} do governo federal \cite{ifi2023}.
    \end{alertblock}
\end{frame}

%% --- Slide 2: Sistema Previdenciário Atual ---
\begin{frame}{O Sistema Previdenciário Atual}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{INSS}
                \begin{itemize}
                    \item Regime de repartição simples
                    \item Alíquotas: 7,5\% a 14\%
                    \item Empregadores: 20\% (folha)
                    \item Déficit: \textbf{R\$ 318,4 bi} (2023)
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{FGTS}
                \begin{itemize}
                    \item 8\% do salário depositado
                    \item Rentabilidade: TR + 3\%/ano
                    \item \textbf{Abaixo da inflação}
                    \item Perda real de até 40\% em 30 anos
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    
    \vspace{0.3cm}
    
    \simplebox{O trabalhador \textbf{não possui propriedade real} sobre suas contribuições \cite{afonso2016}!}
\end{frame}

%% --- Slide 3: Justificativa ---
\subsection{Justificativa e Objetivos}
\begin{frame}{Justificativa}
    \begin{itemize}
        \item Necessidade urgente de reformar o sistema previdenciário brasileiro
        \item Reformas implementadas até o momento têm sido \textbf{paliativas} (ajustes paramétricos)
        \item A tecnologia blockchain oferece oportunidade única de implementar um sistema:
        \begin{itemize}
            \item Transparente e imutável
            \item Descentralizado
            \item Com governança compartilhada
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \successbox{Garantir a propriedade individual dos recursos previdenciários com mecanismos de controle contra uso inadequado.}
\end{frame}

%% --- Slide 4: Objetivos ---
\begin{frame}{Objetivos}
    \textbf{Objetivo Geral:}
    \begin{itemize}
        \item Propor um modelo de mercado financeiro poupador baseado em blockchain como alternativa ao INSS e FGTS
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Objetivos Específicos:}
    \begin{enumerate}
        \item Analisar os problemas estruturais do sistema previdenciário brasileiro atual
        \item Estudar o modelo australiano de \textit{Superannuation} como referência internacional
        \item Propor uma arquitetura de sistema blockchain para gestão previdenciária
        \item Identificar os benefícios e desafios da implementação do modelo proposto
    \end{enumerate}

\section{Fundamentação Teórica}

%% --- Slide 5: Bolsa de Amsterdã ---
\subsection{Origens do Mercado de Capitais}
\begin{frame}{A Bolsa de Amsterdã e a VOC (1602)}
    \begin{itemize}
        \item Primeira \textbf{bolsa de valores} e primeira \textbf{sociedade anônima} do mundo \cite{gelderblom2013}
        \item Solução dos mercadores holandeses: \textbf{pulverização do risco} entre múltiplos investidores
        \item \textbf{Não foi criada por decreto governamental} --- emergiu organicamente das práticas comerciais
    \end{itemize}
    
    \vspace{0.2cm}
    
    \begin{block}{Princípios Fundacionais da VOC}
        \begin{itemize}
            \item \textbf{Responsabilidade limitada}: Acionistas não respondiam pelas dívidas
            \item \textbf{Transferibilidade}: Ações negociadas sem aprovação da empresa
            \item \textbf{Perpetuidade}: Empresa independente da vida dos fundadores
            \item \textbf{Governança corporativa}: Assembleias e conselhos
        \end{itemize}
    \end{block}
\end{frame}

%% --- Slide 6: Primeiro DeFi ---
\begin{frame}{Descentralização sem Blockchain: O Primeiro DeFi}
    Elementos das \textbf{finanças descentralizadas (DeFi)} já presentes na Amsterdã do séc.\ XVII \cite{petram2014}:
    
    \vspace{0.2cm}
    
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Amsterdã (1602)}
                \begin{itemize}
                    \item Negociação direta (\textit{peer-to-peer})
                    \item Liquidez distribuída
                    \item Descoberta de preços por oferta/demanda
                    \item Acesso \textit{permissionless}
                    \item Registro público
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{DeFi Moderno}
                \begin{itemize}
                    \item Smart contracts
                    \item AMMs e pools
                    \item Oráculos de preço
                    \item Carteiras sem KYC
                    \item Blockchain imutável
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    
    \vspace{0.2cm}
    
    \simplebox{A diferença é \textbf{tecnológica}, não conceitual: blockchain substitui confiança por \textbf{verificação criptográfica}.}
\end{frame}

%% --- Slide 7: Democratização ---
\begin{frame}{Democratização do Investimento}
    \begin{itemize}
        \item A VOC permitiu que \textbf{qualquer cidadão} adquirisse ações --- comerciantes, artesãos, viúvas e até servidores domésticos \cite{petram2014}
        \item Ruptura com sistemas restritos a elites mercantis
    \end{itemize}
    
    \vspace{0.2cm}
    
    \begin{exampleblock}{Lições Históricas}
        \begin{itemize}
            \item Poupança popular canalizada para investimentos produtivos gera riqueza coletiva
            \item Mercados líquidos incentivam investimento de longo prazo
            \item Transparência reduz fraudes e aumenta confiança
            \item \textbf{A ausência de controle centralizado não implica em caos}
        \end{itemize}
    \end{exampleblock}
    
    \vspace{0.2cm}
    
    \simplebox{Os mercados de capitais \textbf{nasceram descentralizados}. Blockchain retorna aos princípios fundacionais \cite{dejong2005}.}
\end{frame}

%% --- Slide 8: INSS Detalhado ---
\subsection{Sistema Previdenciário Brasileiro}
\begin{frame}{INSS: Estrutura e Funcionamento}
    \begin{block}{Regime Geral de Previdência Social (RGPS)}
        \begin{itemize}
            \item Opera sob \textbf{regime de repartição simples}: contribuições dos ativos financiam aposentados \cite{gentil2006}
            \item Alíquotas progressivas: \textbf{7,5\% a 14\%} sobre o salário de contribuição
            \item Empregadores (Lucro Real/Presumido): \textbf{20\%} sobre a folha
            \item Simples Nacional: alíquotas reduzidas ou isentas conforme o anexo
        \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Déficit Crescente}
        \begin{itemize}
            \item Déficit do RGPS: \textbf{R\$ 318,4 bilhões} em 2023 \cite{tesouro2024}
            \item Causas: envelhecimento populacional, informalidade, regras desconectadas da capacidade contributiva
        \end{itemize}
    \end{alertblock}
\end{frame}

%% --- Slide 9: FGTS Detalhado ---
\begin{frame}{FGTS: Histórico e Problemas}
    \begin{block}{Estrutura}
        \begin{itemize}
            \item Criado em 1966 como alternativa à estabilidade decenal
            \item Empregadores depositam \textbf{8\% do salário} em conta vinculada
            \item Saque: demissão sem justa causa, aposentadoria, compra de imóvel, doenças graves
        \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Problemas Estruturais}
        \begin{itemize}
            \item Rentabilidade: \textbf{TR + 3\%/ano} --- historicamente abaixo da inflação \cite{afonso2016}
            \item Em 30 anos, perda real de até \textbf{40\%} do valor \cite{insper2019}
            \item Recursos usados pelo governo para financiar programas habitacionais e infraestrutura
            \item Transferência de riqueza do trabalhador sem remuneração adequada
        \end{itemize}
    \end{alertblock}
\end{frame}

%% --- Slide 10: Impacto Fiscal ---
\begin{frame}{Impacto Fiscal e Sustentabilidade}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{alertblock}{Números Atuais}
                \begin{itemize}
                    \item Previdência consome \textbf{$\sim$13\% do PIB}
                    \item RGPS + RPPS + assistência: \textbf{54\% das despesas primárias} \cite{ifi2023}
                    \item Projeção sem reforma: \textbf{$>$20\% do PIB} nas próximas décadas \cite{tafner2019}
                \end{itemize}
            \end{alertblock}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{block}{Círculo Vicioso}
                \begin{enumerate}
                    \item Rigidez orçamentária
                    \item Falta de investimento público
                    \item Baixa produtividade
                    \item Menor arrecadação
                    \item Agrava o déficit
                \end{enumerate}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

%% --- Slide 11: Modelo Australiano ---
\subsection{Modelo Australiano}
\begin{frame}{Modelo Australiano de Superannuation}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item Sistema de \textbf{capitalização individual obrigatória} desde 1992 \cite{bateman2001}
                \item Empregadores contribuem com \textbf{11\% do salário} (previsão: 12\%)
                \item Trabalhador escolhe fundo e estratégia de investimento
                \item Acesso apenas na \textbf{idade de preservação} (55--60 anos)
                \item Desafios: cobertura de informais, educação financeira, risco de mercado
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{exampleblock}{Resultados (30 anos)}
                \begin{itemize}
                    \item \textbf{3,5 trilhões AUD} acumulados
                    \item \textbf{$\sim$170\% do PIB} australiano
                    \item Rentabilidade média superior à inflação
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}
    
    \vspace{0.3cm}
    
    \successbox{Demonstra que sistemas de capitalização individual \textbf{podem ser implementados com sucesso}.}
\end{frame}

%% --- Slide 12: Fundamentos Criptográficos ---
\subsection{Tecnologia Blockchain}
\begin{frame}{Fundamentos Criptográficos}
    \begin{block}{Problema dos Generais Bizantinos \cite{lamport1982}}
        Como múltiplos nós alcançam \textbf{consenso} sobre o estado verdadeiro de um sistema, mesmo com nós comprometidos? Considerado impossível em redes abertas --- até o Bitcoin.
    \end{block}
    
    \vspace{0.2cm}
    
    \begin{block}{Hashcash e Prova de Trabalho \cite{back2002}}
        Exige trabalho computacional para criar blocos. Propriedades da função \textit{hash}: determinística, unidirecional, sensível a mudanças, resistente a colisões.
    \end{block}
    
    \vspace{0.2cm}
    
    \begin{block}{Problema do Gasto Duplo}
        Bitcoin combina \textit{Proof of Work} + cadeia de blocos para resolver conflitos de forma determinística, sem autoridade central. Regra: \textbf{maior trabalho acumulado} vence.
    \end{block}
\end{frame}

%% --- Slide 13: Chaves e HD Wallets ---
\begin{frame}{Chaves Assimétricas e HD Wallets}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Chaves Criptográficas}
                \begin{itemize}
                    \item \textbf{Chave Privada}: 256 bits secretos --- controle absoluto dos fundos
                    \item \textbf{Chave Pública}: derivada por curva elíptica --- compartilhável
                    \item \textbf{Assinatura digital}: prova posse sem revelar chave privada
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{HD Wallets (BIP-32/39) \cite{bip32,bip39}}
                \begin{itemize}
                    \item Semente mnemônica: \textbf{12 ou 24 palavras}
                    \item Árvore infinita de chaves derivadas
                    \item Backup simplificado
                    \item Privacidade: novo endereço por transação
                    \item Carteiras frias (\textit{offline})
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

%% --- Slide 14: Características da Blockchain ---
\begin{frame}{Características Arquiteturais da Blockchain}
    \begin{block}{Principais Características \cite{nakamoto2008}}
        \begin{itemize}
            \item \textbf{Descentralização}: Dados replicados em múltiplos nós, eliminando pontos únicos de falha
            \item \textbf{Imutabilidade}: Transações registradas não podem ser alteradas ou excluídas
            \item \textbf{Transparência ou Privacidade}: Dependendo da implementação --- transações públicas ou valores/destinatários ocultos
            \item \textbf{Segurança criptográfica}: Proteção por algoritmos robustos
        \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{exampleblock}{Tokenização de Ativos}
        Representar direitos de propriedade sobre ativos reais em tokens na blockchain --- permite fracionamento de investimentos e aumento da liquidez.
    \end{exampleblock}
\end{frame}

%% --- Slide 16: Curva de Laffer ---
\subsection{Soluções para Dívida Previdenciária}
\begin{frame}{Curva de Laffer e Dignidade do Trabalhador}
    \begin{itemize}
        \item Qualquer reforma deve \textbf{evitar desânimo social} --- decisões burocráticas não podem ser vistas como expropriação
        \item Assim como a \textbf{Curva de Laffer} demonstra um ponto ótimo de tributação \cite{laffer2004}, existe um ponto ótimo de \textit{motivação do trabalhador}
    \end{itemize}
    
    \vspace{0.2cm}
    
    \begin{alertblock}{Risco}
        Se o trabalhador percebe que contribuições serão confiscadas/diluídas, o incentivo para trabalhar formalmente \textbf{diminui drasticamente}.
    \end{alertblock}
    
    \vspace{0.2cm}
    
    \begin{exampleblock}{Princípio Fundamental}
        O primeiro passo da reforma deve ser \textbf{pagar, com juros e correção}, os depósitos de todos os brasileiros vivos, convertidos em ativos no novo sistema. \textit{Custe o que custar, o país deve estar à venda.}
    \end{exampleblock}
\end{frame}

%% --- Slide 21: Princípios ---
\subsection{Visão Geral}
\begin{frame}{Princípios Fundamentais}
    \begin{exampleblock}{Visão Geral}
        Mercado financeiro poupador que substitui gradualmente INSS e FGTS, permitindo que trabalhadores invistam diretamente em empresas brasileiras.
    \end{exampleblock}

        \item \textbf{Transparência nas regras}: Código auditável, privacidade nas transações
        \item \textbf{Automatização}: Regras executadas por smart contracts
    \end{itemize}
\end{frame}

            \item \textbf{Convergência}: As frentes se encontram na faixa 40--50 anos (maiores saldos)
        \end{itemize}
    \end{exampleblock}
\end{frame}
