\section{Introdução}
\begin{frame}{Contextualização}
    \begin{itemize}
        \item O Brasil enfrenta \textbf{desafios econômicos estruturais} que limitam seu desenvolvimento há décadas
        \item O foco excessivo em corrupção e polarização desvia a atenção dos problemas verdadeiramente estruturantes:
        \begin{itemize}
            \item Infraestrutura deficiente
            \item Baixa taxa de poupança nacional
            \item Crédito historicamente caro
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Problema Central}
        Os gastos com previdência e assistência social representam aproximadamente \textbf{54\% das despesas primárias} do governo federal.
    \end{alertblock}
\end{frame}

%% ---------------------------------------------------------------------------
\begin{frame}{O Sistema Previdenciário Atual}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{INSS}
                \begin{itemize}
                    \item Regime de repartição simples
                    \item Déficit de R\$ 318,4 bi (2023)
                    \item Trabalhadores ativos financiam aposentados
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{FGTS}
                \begin{itemize}
                    \item 8\% do salário depositado
                    \item Rentabilidade inferior à inflação
                    \item Perda real de até 40\% em 30 anos
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    
    \vspace{0.4cm}
    
    \simplebox{O trabalhador \textbf{não possui propriedade real} sobre suas contribuições!}
\end{frame}

%% ---------------------------------------------------------------------------
\begin{frame}{Teorema do Júri de Condorcet}
    \begin{block}{Analogia: Júri Popular vs.\ Juiz Singular}
        No \textbf{júri popular}, a deliberação coletiva de múltiplos indivíduos independentes
        tende a produzir decisões mais acuradas do que o julgamento singular de um magistrado \cite{condorcet1785}.
    \end{block}

    \vspace{0.3cm}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Mão Invisível do Mercado}
                \begin{itemize}
                    \item Milhões de agentes econômicos alocam capital de forma autônoma
                    \item Descoberta de preços mais eficiente que gestão centralizada \cite{hayek1945}
                \end{itemize}
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Implicação}
                \begin{itemize}
                    \item Decisões descentralizadas $>$ decisões centralizadas
                    \item Fundamenta a proposta de gestão individual dos recursos previdenciários
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\subsection{Criptografia}
\begin{frame}{Funções Hash Criptográficas}
    \begin{block}{Definição}
        Função unidirecional $H(m) = x$ que mapeia entrada de tamanho arbitrário para saída de tamanho fixo \cite{tomaz2014}.
    \end{block}

    \vspace{0.2cm}

    \begin{table}
        \centering
        \small
        \begin{tabular}{p{4cm}p{5.5cm}}
            \toprule
            \textbf{Propriedade} & \textbf{Significado} \\
            \midrule
            Resistência à pré-imagem & Dado $x$, inviável encontrar $m$ \\
            Res.\ à 2ª pré-imagem & Dado $m$, inviável achar $m' \neq m$ com $H(m')=H(m)$ \\
            Resistência à colisão & Inviável achar \textit{qualquer} par $(m, m')$ colidindo \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.2cm}

    \begin{exampleblock}{Paradoxo do Aniversário}
        $\bar{p}(n) = \dfrac{365!}{365^n \cdot (365-n)!}$ \quad --- \quad Com 23 pessoas, $P(\text{colisão}) > 50\%$.
        Para SHA-256: necessárias $2^{128}$ tentativas (ataque do aniversário).
    \end{exampleblock}
\end{frame}

%% ---------------------------------------------------------------------------
\begin{frame}{Criptossistema RSA}
    \begin{block}{Geração de Chaves (5 passos) \cite{rivest1978, tomaz2014}}
        \begin{enumerate}
            \item Escolher primos grandes $p$ e $q$
            \item Calcular módulo $n = p \times q$
            \item Calcular totiente $\phi(n) = (p-1)(q-1)$
            \item Escolher $e$ tal que $\text{mdc}(e, \phi(n)) = 1$
            \item Calcular $d$: \; $e \cdot d \equiv 1 \pmod{\phi(n)}$ \; (alg.\ euclidiano estendido)
        \end{enumerate}
    \end{block}

    \vspace{0.3cm}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Cifragem / Decifragem}
                $C = M^e \bmod n$ \\[4pt]
                $M = C^d \bmod n$
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Assinatura Digital}
                \text{Assinar}: $S = H(M)^d \bmod n$ \\[4pt]
                \text{Verificar}: $H(M) = S^e \bmod n$ \\[4pt]
                {\small Garante: integridade, autenticidade, não repúdio}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

%% ---------------------------------------------------------------------------
\begin{frame}{Assinaturas em Anel (\textit{Ring Signatures})}
    \begin{block}{Conceito \cite{rivest2001}}
        Permite assinar uma mensagem provando pertencer a um \textbf{grupo}, sem revelar \textbf{quem} assinou.
    \end{block}

    \vspace{0.2cm}

    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{enumerate}
                \item \textbf{Preparação}: calcula $z = H(m)$, sorteia $v$
                \item \textbf{Preenchimento}: cifra valores com chaves públicas dos demais
                \item \textbf{Montagem}: encadeia valores em circuito circular
                \item \textbf{Fechamento}: usa \textit{chave privada} para fechar o anel
            \end{enumerate}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{exampleblock}{Propriedades}
                \begin{itemize}
                    \item Sem gerente central
                    \item Sem configuração prévia
                    \item Grupo variável a cada assinatura
                    \item \textbf{Resgate de autoria} possível \cite{tomaz2014}
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \vspace{0.2cm}

    \simplebox{Aplicação: trabalhador prova ser participante legítimo \textbf{sem revelar identidade} --- protege saldo e movimentações.}
\end{frame}

            \item \textcolor{green_ufc}{\textbf{\checkmark}} Governo + Auditoria --- recuperação de acesso
            \item \textcolor{red}{\textbf{\texttimes}} Trabalhador + Auditoria --- \textbf{não autorizada}
